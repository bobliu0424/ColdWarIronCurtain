	RPR_gui_main = {
		context_type = player_context

		}

		effects = {
			Percentage_checkbox_click = {
				if = {
					limit = {
						check_variable = { show_percentage = 1 }
					}
					set_variable = { show_percentage = 2 }
				}
				else = {
					set_variable = { show_percentage = 1 }
				}
				if = {
					limit = {
						check_variable = { ROOT.refresh_show_french_gui = 1 }
					}
					set_variable = { ROOT.refresh_show_french_gui = 0 }
				}
				else = {
					set_variable = { ROOT.refresh_show_french_gui = 1 }
				}
			}
		}
		properties = {
			Percentage_checkbox = {
 				frame = show_percentage
			}
